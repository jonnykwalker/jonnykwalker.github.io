Full Group Membership of Multiple Users 

#Purpose: Find the Active Directory group membership for a list of users and export them to individual text files.

$csvFilePath = 'C:\Users\Public\Documents\Users_to_Check.csv'  

 

$csvFile = Import-Csv -Path $csvFilePath  

   

ForEach ($row in $csvFile) {  

$samaccount = $row.samaccountname  

 

$user = Get-ADPrincipalGroupMembership $samaccount | Select name | Out-File "C:\Users\Public\Documents\Group_Check\$($row.samaccountname).txt"

}
